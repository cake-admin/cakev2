# Colors

cake& colors are semantic tokens from the Figma variable system
(`&color.<group>.<leaf>`). They describe a role — such as a surface, border,
or destructive action — rather than a raw color choice. That lets the same
component adapt across Light A, Dark A, and Windows High Contrast.

Use the **Theme** toolbar to inspect the active mode. The reference below
always shows all three resolved values, while the surrounding documentation
re-themes live with the toolbar.

export const groups = [...new Set(tokens.list.map((t) => t.group))];

export const groupId = (name) => `color-${name.replace(/[^a-z0-9]+/gi, '-').toLowerCase()}`;

export const MODE_LABELS = {
  lightA: 'Light A',
  darkA: 'Dark A',
  winHct: 'Win HCT',
};

export const MUTED = 'var(--color-text-icon-secondary)';
export const BORDER = 'var(--color-stroke-border)';
export const CONTAINER = 'var(--color-surfaces-container)';

export const Swatch = ({ color, label }) => (
  <span
    aria-label={`${label}: ${color}`}
    title={`${label}: ${color}`}
    style={{
      display: 'block',
      width: '100%',
      height: 48,
      borderRadius: 8,
      background: color,
      border: `1px solid ${BORDER}`,
      boxSizing: 'border-box',
    }}
  />
);

export const ModeValue = ({ mode, color }) => (
  <div
    style={{
      display: 'grid',
      gridTemplateColumns: 'minmax(48px, 1fr)',
      gap: 6,
      minWidth: 0,
    }}
  >
    <Swatch color={color} label={MODE_LABELS[mode]} />
    <div style={{ color: MUTED, fontSize: 11, lineHeight: 1.3 }}>{MODE_LABELS[mode]}</div>
    <code style={{ fontSize: 11, overflowWrap: 'anywhere' }}>{color}</code>
  </div>
);

export const TokenCard = ({ token }) => (
  <article
    style={{
      background: CONTAINER,
      border: `1px solid ${BORDER}`,
      borderRadius: 12,
      padding: 16,
      display: 'grid',
      gap: 16,
    }}
  >
    <div style={{ display: 'grid', gap: 4 }}>
      <code style={{ fontSize: 13, fontWeight: 600, overflowWrap: 'anywhere' }}>
        color.{token.groupKey}.{token.leafKey}
      </code>
      <span style={{ color: MUTED, fontSize: 12, overflowWrap: 'anywhere' }}>{token.figmaPath}</span>
    </div>
    <div
      style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(3, minmax(0, 1fr))',
        gap: 10,
      }}
    >
      <ModeValue mode="lightA" color={token.lightA} />
      <ModeValue mode="darkA" color={token.darkA} />
      <ModeValue mode="winHct" color={token.winHct} />
    </div>
  </article>
);

export const Group = ({ name }) => (
  <section style={{ marginTop: 48 }}>
    <h2 id={groupId(name)}>
      <code>{name}</code>
    </h2>
    <p style={{ color: MUTED, marginTop: -8, marginBottom: 20 }}>
      {tokens.list.filter((t) => t.group === name).length} semantic tokens
    </p>
    <div
      style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(auto-fit, minmax(290px, 1fr))',
        gap: 16,
      }}
    >
      {tokens.list
        .filter((t) => t.group === name)
        .map((t) => (
          <TokenCard key={t.figmaPath} token={t} />
        ))}
    </div>
  </section>
);

## Usage

```tsx
import styled from 'styled-components';

const Card = styled.div`
  background: var(--color-surfaces-container);
  color: var(--color-text-icon-primary);
  border: var(--stroke-100) solid var(--color-stroke-border);
`;
```
