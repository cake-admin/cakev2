# Colors

cake& colors are semantic tokens from the Figma variable system
(`&color.<group>.<leaf>`). They describe a role — such as a surface, border,
or destructive action — rather than a raw color choice. That lets the same
component adapt across Light A, Dark A, and Windows High Contrast.

Use the **Theme** toolbar to inspect the active mode. The reference below
always shows all three resolved values, while the surrounding documentation
re-themes live with the toolbar.

export const groups = [...new Set(tokens.list.map((t) => t.group))];

export const groupId = (name) => `color-${name.replace(/[^a-z0-9]+/gi, '-').toLowerCase()}`;

export const MODE_LABELS = {
  lightA: 'Light A',
  darkA: 'Dark A',
  winHct: 'Win HCT',
};

export const MUTED = 'var(--color-text-icon-secondary)';
export const BORDER = 'var(--color-stroke-border)';
export const CONTAINER = 'var(--color-surfaces-container)';

export const MODE_EXPORT_KEYS = {
  lightA: 'light.a',
  darkA: 'dark.a',
  winHct: 'win hct',
};

export const toFigmaColor = (value) => {
  if (value == null || value === '') return null;

  const hex = value.match(/^#([\da-f]{6})$/i);
  if (hex) {
    const number = parseInt(hex[1], 16);
    return {
      r: ((number >> 16) & 255) / 255,
      g: ((number >> 8) & 255) / 255,
      b: (number & 255) / 255,
      a: 1,
    };
  }

  const rgba = value.match(/^rgba?\(([^)]+)\)$/i);
  if (!rgba) throw new Error(`Unsupported color value: ${value}`);

  const parts = rgba[1].split(',').map((part) => Number(part.trim()));
  const [r, g, b, a = 1] = parts;
  return { r: r / 255, g: g / 255, b: b / 255, a };
};

export const downloadJson = (filename, data) => {
  const url = URL.createObjectURL(
    new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' })
  );
  const link = document.createElement('a');
  link.href = url;
  link.download = filename;
  link.style.display = 'none';
  document.body.appendChild(link);
  link.click();
  link.remove();
  window.setTimeout(() => URL.revokeObjectURL(url), 1000);
};

export const exportThemes = () => {
  const collectionId = 'VariableCollectionId:cake-color';
  const modes = Object.entries(MODE_EXPORT_KEYS).map(([key, modeId]) => ({
    modeId,
    name: MODE_LABELS[key],
  }));

  const variables = tokens.list
    .map((token, index) => {
      const valuesByMode = Object.fromEntries(
        Object.entries(MODE_EXPORT_KEYS)
          .map(([key, modeId]) => {
            const color = toFigmaColor(token[key]);
            return color ? [modeId, color] : null;
          })
          .filter(Boolean)
      );

      if (Object.keys(valuesByMode).length === 0) return null;

      return {
        id: `VariableID:cake-color:${index + 1}`,
        name: token.figmaPath.replace(/^&color\//, ''),
        description: `cake& semantic color: ${token.figmaPath}`,
        resolvedType: 'COLOR',
        variableCollectionId: collectionId,
        valuesByMode,
        codeSyntax: {
          WEB: `--color-${token.groupKey.replace(/[A-Z]/g, (letter) => `-${letter.toLowerCase()}`)}-${token.leafKey.replace(/[A-Z]/g, (letter) => `-${letter.toLowerCase()}`)}`,
        },
      };
    })
    .filter(Boolean);

  const file = {
    meta: {
      format: 'figma-variables-export',
      formatVersion: '1.0.0',
      source: 'cake& design system',
      exportedAt: new Date().toISOString(),
    },
    collections: [
      {
        id: collectionId,
        name: '&color',
        modes,
        defaultModeId: MODE_EXPORT_KEYS.lightA,
        variableIds: variables.map((variable) => variable.id),
      },
    ],
    variables,
  };

  downloadJson('cake-color-themes.json', file);
};

export const Swatch = ({ color, label }) => (
  <span
    aria-label={`${label}: ${color}`}
    title={`${label}: ${color}`}
    style={{
      display: 'block',
      width: '100%',
      height: 48,
      borderRadius: 8,
      background: color,
      border: `1px solid ${BORDER}`,
      boxSizing: 'border-box',
    }}
  />
);

export const ModeValue = ({ mode, color }) => (
  <div
    style={{
      display: 'grid',
      gridTemplateColumns: 'minmax(48px, 1fr)',
      gap: 6,
      minWidth: 0,
    }}
  >
    <Swatch color={color} label={MODE_LABELS[mode]} />
    <div style={{ color: MUTED, fontSize: 11, lineHeight: 1.3 }}>{MODE_LABELS[mode]}</div>
    <code style={{ fontSize: 11, overflowWrap: 'anywhere' }}>{color}</code>
  </div>
);

        color.{token.groupKey}.{token.leafKey}

      style={{

    </p>
    <div
      style={{
        display: 'grid',
        gridTemplateColumns: 'repeat(auto-fit, minmax(290px, 1fr))',

);
